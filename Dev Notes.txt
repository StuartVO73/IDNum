public class HolidayResponse {
    
    public HolidayResponse (String response) {
        Map<String, Object> results = (Map<String, Object>) JSON.deserializeUntyped(response);
        //system.debug (results.get('meta'));
        Holiday[] holidayList = new List<Holiday>();
        Object[] holidaysContainer = (Object[])((Map<String, Object>)results.get('response')).get('holidays');
        system.debug(holidaysContainer);
        for (Object H : holidaysContainer) {
            holidayList.add (new Holiday (H));
        }
        for (Holiday H : holidayList) {
            system.debug (H);
        }
    }   
}

public class Holiday {
    String name;
    String isoDate;
    String description;
    
    public Holiday (Object H) {
        Map<String, Object> H2 = (Map<String, Object>) H;
        this.name = (String) H2.get('name');
	    this.description = (String) H2.get('description');
    	this.isoDate = (String) ((Map <String, Object>) H2.get('date')).get('iso');
    }
    
    public override String toString () {
        return name + ', ' + description + ', ' + isoDate;
    }
